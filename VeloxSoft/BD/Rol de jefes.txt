-- Crear rol con límite de 5 conexiones
CREATE ROLE admin WITH LOGIN PASSWORD 'fruteria7' CONNECTION LIMIT 5;

-- Conexión y acceso al esquema
GRANT CONNECT ON DATABASE "Fruteria7" TO admin;
GRANT USAGE ON SCHEMA public TO admin;

-- Permisos sobre TABLAS (CRUD completo: Select, Insert, Update, Delete)
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO admin;
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO admin;

-- Permisos sobre SECUENCIAS (Control total)
GRANT USAGE, SELECT, UPDATE ON ALL SEQUENCES IN SCHEMA public TO admin;
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT USAGE, SELECT, UPDATE ON SEQUENCES TO admin;
