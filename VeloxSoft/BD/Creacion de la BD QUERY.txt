-- =======================================
-- CREACIÓN DE TABLAS BASE DEL SISTEMA
-- =======================================

-- Tabla de Colonias
CREATE TABLE TBL_COLONIA (
    idcolonia INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Colonia VARCHAR(100) NOT NULL
);

-- Tabla de Direcciones
CREATE TABLE TBL_DIRECCION (
    id_direc INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Num_casa VARCHAR(45),
    Calle VARCHAR(45) NOT NULL,
    Cruzamientos VARCHAR(45) NOT NULL,
    Referencia VARCHAR(200) NOT NULL,
    Colonia_d INTEGER NOT NULL,
	FOREIGN KEY (Colonia_d) REFERENCES TBL_COLONIA(idcolonia)
);
-- Tabla de Clientes
CREATE TABLE TBL_CLIENTE (
    ID_cel BIGINT  PRIMARY KEY,
    Nombre VARCHAR(50) NOT NULL,
    Apellido VARCHAR(50) NOT NULL,
    Apodo VARCHAR(30),
    Direccion_c INTEGER,
    FOREIGN KEY (Direccion_c) REFERENCES TBL_DIRECCION(id_direc)
);


-- Tabla de Fechas
CREATE TABLE TBL_FECHA (
    ID_fecha DATE PRIMARY KEY,
    Dia VARCHAR(2) NOT NULL,
    Mes VARCHAR(2) NOT NULL,
    Anio VARCHAR(4) NOT NULL
);

-- Tabla de Tipos de Pago
CREATE TABLE TBL_PAGO (
    ID_pago VARCHAR(2) PRIMARY KEY,
    Tipo_pago VARCHAR(45) NOT NULL
);


-- Tabla de Estado del Pedido
CREATE TABLE TBL_ESTADO (
    ID_estado VARCHAR(10) PRIMARY KEY,
    Tipo_Estado VARCHAR(45) NOT NULL
);


CREATE TYPE rol AS ENUM ('0', '1', '2', '3');
CREATE TABLE tbl_usuario (
    	id BIGINT PRIMARY KEY,
	nombre VARCHAR(50) NOT NULL,
	password VARCHAR(250) NOT NULL,
	tipo rol NOT NULL,
	secion VARCHAR(50) NOT NULL,
	estado VARCHAR(50) NOT NULL
);



-- Tabla de Gastos
CREATE TABLE TBL_GASTO (
    id_gasto BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    descripcion VARCHAR(255) NOT NULL
);

-- Tabla de Corte
CREATE TABLE TBL_CORTE (
	id_corte BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
	fecha DATE NOT NULL, 
	total_ventas DECIMAL(10,4) NOT NULL,
	total_gasto DECIMAL(10,4) NOT NULL,
	ganancia DECIMAL(10,4) NOT NULL
);


-- Tabla de detalle gasto
CREATE TABLE TBL_DET_GASTO (
	id_gasto_d BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
	monto DECIMAL(10,4) NOT NULL,
	fecha DATE NOT NULL,
	id_gasto BIGINT NOT NULL,
	id_corte BIGINT,
	FOREIGN KEY (id_gasto) REFERENCES TBL_GASTO(id_gasto),
	FOREIGN KEY (id_corte) REFERENCES TBL_CORTE(id_corte)
);



-- Tabla de Productos
CREATE TABLE TBL_PRODUCTO (
    ID_producto VARCHAR(30) PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL,
    Cantidad DECIMAL(10,4) NOT NULL,
    Precio DECIMAL(10,4) NOT NULL,
    estado BOOLEAN DEFAULT TRUE
);


-- Tabla de Ventas
CREATE TABLE TBL_VENTA (
    id_venta BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Cantidad DECIMAL(10,4) NOT NULL,
    Importe_g DECIMAL(10,4) NOT NULL,
    Fecha DATE,
    Num_cel BIGINT,
    Tipo_pago VARCHAR(2),
    Tipo_estado VARCHAR(10),
    id_corte BIGINT,
    id_usuario BIGINT NOT NULL,
    FOREIGN KEY (Fecha) REFERENCES TBL_FECHA(ID_fecha),
    FOREIGN KEY (Num_cel) REFERENCES TBL_CLIENTE(ID_cel),
    FOREIGN KEY (Tipo_pago) REFERENCES TBL_PAGO(ID_pago),
    FOREIGN KEY (Tipo_estado) REFERENCES TBL_ESTADO(ID_estado),
    FOREIGN KEY (id_corte) REFERENCES TBL_CORTE(id_corte),
    FOREIGN KEY (id_usuario) REFERENCES tbl_usuario(id)
);

-- Tabla Detalle del Pedido
CREATE TABLE TBL_DETALLE_VENTAS (
    ID_detalle_venta SERIAL PRIMARY KEY,
    Cantidad INT NOT NULL,
    Importe_p DECIMAL(10,4) NOT NULL,
    nventa BIGINT,
    idprod VARCHAR(30),
    FOREIGN KEY (nventa) REFERENCES TBL_VENTA(id_venta),
    FOREIGN KEY (idprod) REFERENCES TBL_PRODUCTO(ID_producto)
);
